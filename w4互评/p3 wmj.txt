#include<iostream>
#include<string>
using namespace std;
template<class T>
void sort(T *p,int size)
{
 T temp;
 //√∞≈›≈≈–Ú
 for (int i = 0; i < size; i++)
 {
  for (int j = 0; j < i; j++)
  {
   if (p[j] > p[i])
   {
    temp = p[i];
    p[i] = p[j];
    p[j] = temp;
   }
  }
 }
}
template<class T>
void input(T *p, int size)
{
 for (int i = 0; i < size; i++)
  cout << p[i] << " ";
 cout << endl;
}
int main()
{
 //int–Õ
 int array1[10] = { 1,5,78,14,48,15,65,15,4,89 };
 sort(array1, 10);
 input(array1, 10);
 //double–Õ
 float array2[10] = { 1.2,5.4,6.2,0.1,9.2,5.7,10.0,9.2,12.0,36.1 };
 sort(array2, 10);
 input(array2, 10);
 //string–Õ
 string array3[10] =
 {
  "wangwangwang","goodmorning","goodafternoon","c++","happy",
  "loving","spring","fight","unity","wang"
 };
 sort(array3, 10);
 input(array3, 10);
 return 0;
}
